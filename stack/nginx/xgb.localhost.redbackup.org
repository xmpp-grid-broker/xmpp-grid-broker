upstream openfire_http_bind {
    server openfire:7070 fail_timeout=0;
}

upstream xgb_development_server {
  server angular:4200 fail_timeout=0;
}

server {
    listen 443 ssl http2;
    server_name xgb.localhost.redbackup.org;

    ssl_certificate /etc/dehydrated/certs/xgb.localhost.redbackup.org/fullchain.pem;
    ssl_certificate_key /etc/dehydrated/certs/xgb.localhost.redbackup.org/privkey.pem;
    include /etc/nginx/tls-include.conf;

    ssl_verify_client on;
    ssl_client_certificate /etc/dehydrated/ca-crt.pem;

    access_log /var/log/nginx/${host}.access.log;
    error_log /var/log/nginx/${host}.error.log info;

    keepalive_timeout 6000;

    # support up to ~20MB uploads
    client_max_body_size 20M;

    # security headers
    add_header Content-Security-Policy "default-src 'none'; script-src 'self' 'unsafe-inline' 'unsafe-eval'; connect-src 'self'; img-src 'self'; style-src 'self' 'unsafe-inline';";
    add_header 'Access-Control-Allow-Origin' 'xgb.localhost.redbackup.org';

    # Openfire BOSH
    location /http-bind {
        if ($request_method = 'OPTIONS') {
            add_header 'Content-Type' 'text/plain; charset=utf-8';
            add_header 'Content-Length' 0;
            return 204;
        }
        proxy_set_header Host localhost.redbackup.org;
        proxy_set_header X-Forwarded-Host localhost.redbackup.org;
        proxy_set_header X-Forwarded-Server localhost.redbackup.org;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

        proxy_pass http://openfire_http_bind/http-bind;
        break;
    }

    # Openfire WebSocket
    location /ws {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";

        proxy_set_header Host localhost.redbackup.org;
        proxy_set_header X-Forwarded-Host localhost.redbackup.org;
        proxy_set_header X-Forwarded-Server localhost.redbackup.org;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

        proxy_pass http://openfire_http_bind/ws;
        break;
    }

    # Live-reload WebSockets for ng serve
    location /sockjs-node/ {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";

        proxy_set_header Host $http_host;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Server $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_redirect off;

        proxy_pass http://xgb_development_server;
        break;
    }

    # Angular from ng serve
    location / {
        proxy_set_header Host $http_host;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Server $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_redirect off;

        proxy_pass http://xgb_development_server;
        break;
    }

#     location / {

#         index  index.html index.htm;
#     }


}
